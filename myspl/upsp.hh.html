class UPSPHeader: public Message
{
  public:
    UPSPHeader():
      Message(4,"HEADER")
    {
      // Header fields
      assign(0,new Int4Field("length"));
      assign(1,new Int2Field("msg_type"));
      assign(2,new Int4Field("client_data_1"));
      assign(3,new Int4Field("client_data_2"));
    }
};
