# login
#
#   output:
#     0 (=> ok)
#     1 (=> timeout)
#     2 (=> user authorization error)
#     3 (=> other user already logged in)
#
proc login {username password} {
  ctrla
  expect *

  send "LOGIN\r"

  expect {
    "USERNAME:~" {
      send "$username\r"
      exp_continue
    }
    "PASSWORD:~" {
      send "$password\r"
      exp_continue
    }
    "LOGIN COMPLETE" {
      return 0
    }
    "USER AUTHORIZATION ERROR" {
      return 2
    }
    "OTHER USER ALREADY LOGGED IN" {
      return 3
    }
  }

  return 1
}

#
# logout
#
#   output:
#     0 (=> ok)
#     1 (=> timeout)
#     2 (=> user not logged in)
#
proc logout {} {
  ctrla
  expect *

  send "LOGOUT\r"

  expect {
    "LOGOUT COMPLETE" {
      return 0
    }
    "USER NOT LOGGED IN" {
      return 2
    }
  }

  return 1
}

# ctrla
#
#   output:
#     0 (=> ok)
#     1 (=> timeout)
#
proc ctrla {} {
  send [binary format H2 01]\r

  expect {
    ">" {
      return 0
    }
  }

  return 1
}

# assignnpentry
#
#   output:
#     0 (=> ok)
#     1 (=> timeout)
#     2 (=> user not logged in)
#     3 (=> this range is already in the database)
#
proc assignnpentry {subscribertype subscribernumber routingnumber} {
  # remove "39" from subscriber number(s)
  regsub -- "^39" $subscribernumber "" subscribernumber
  regsub -- "-39" $subscribernumber "-" subscribernumber

  ctrla
  expect *

  send "ASSIGN NP ENTRY\r"

  expect {
    "SUBSCRIBER TYPE=~" {
      send "$subscribertype\r"
      exp_continue
    }
    "SUBSCRIBER NUMBER=~" {
      send "$subscribernumber\r"
      exp_continue
    }
    "ROUTING NUMBER=~" {
      send "$routingnumber\r"
      exp_continue
    }
    "OR ABORT" {
      send "Y\r"
      exp_continue
    }
    "UPDATED" {
      return 0
    }
    "USER NOT LOGGED IN" {
      return 2
    }
    "ALREADY" {
      # ... "THIS RANGE IS ALREADY IN THE DATABASE" ...
      return 3
    }
  }

  return 1
}
