package it.mysvc.myexpr;

public class Token
{
  enum Type
  {
    Operand,
    UnaryOperator,
    BinaryOperator,
    OpenedParenthesis,
    ClosedParenthesis
  }

  public Type type;

  public int code;

  public String info;

  public Token()
  {
  }

  public Token(Type type,int code)
  {
    this.type = type;
    this.code = code;
  }

  public Token(Type type,int code,String info)
  {
    this.type = type;
    this.code = code;
    this.info = info;
  }

  public Type getType()
  {
    return type;
  }

  public int getCode()
  {
    return code;
  }

  public String getInfo()
  {
    return info;
  }
}
